CREATION
-----------

CREATE TABLE HELP_CENTER(
	ID NUMBER,
	COMPLAINT VARCHAR2 100,
	EMAIL VARCHAR2 20,
	DATE VARCHAR2 20,
	ADMIN_ID NUMBER,
	PRIMARY KEY (ID),
	FOREIGN KEY (EMAIL) REFERENCES USERS(EMAIL),
	FOREIGN KEY (ADMIN_ID) REFERENCES ADMINS(ID));

CREATE TABLE ADMIN(
	ID NUMBER,
	NAME VARCHAR2 20,
	EMAIL VARCHAR2 20,
	PASSWORD VARCHAR2 20,
	PRIMARY KEY (ID));

CREATE TABLE HOTELS(
	HOTEL_NAME VARCHAR2 20,
	RATING NUMBER,
	LOCATION VARCHAR2 20,
	SPONSOR_NAME VARCHAR2 20,
	PRIMARY KEY (HOTEL_NAME),
	FOREIGN KEY (SPONSOR_NAME) REFERENCES SPONSORS(NAME));

CREATE TABLE REVIEWES(
	ID NUMBER,
	TEXT VARCHAR2 200,
	DATE VARCHAR2 20,
	EMAIL VARCHAR2 20,
	RATING NUMBER,
	SPONSOR_NAME VARCHAR2 20,
	PRIMARY KEY (ID),
	FOREIGN KEY (EMAIL) REFERENCES USERS(EMAIL),
	FOREIGN KEY (SPONSOR_NAME) REFERENCES SPONSORS(NAME));

CREATE TABLE SPONSORS(
	NAME VARCHAR2 20,
	URL VARCHAR2 20,
	PRIORITY NUMBER, 
	PRIMARY KEY (NAME));

CREATE TABLE USERS(
	NAME VARCHAR2 20,
	EMAIL VARCHAR2 20,
	PASSWORD VARCHAR2 20,
	NATIONALITY VARCHAR2 20,
	PRIMARY KEY (EMAIL));

		PROCEDURE RETURN_RATING (hname in VARCHAR2 ,hrating OUT NUMBER)
